<div id="scoreBox">Score: 0</div>

<script>
// Cat image files + points
const cats = [
    { src: "Cat.defoult.dance.gif?v=2", points: 2 },
    { src: "Cat.fire.dance.gif?v=2", points: 10 },
    { src: "Cat.holy.dance.gif?v=2", points: 25 }
];

let score = 0;
const scoreBox = document.getElementById("scoreBox");

function spawnCat() {
    const catData = cats[Math.floor(Math.random() * cats.length)];
    const cat = document.createElement("img");

    cat.src = catData.src;
    cat.className = "cat";

    cat.style.left = Math.random() * (window.innerWidth - 200) + "px";
    cat.style.top = Math.random() * (window.innerHeight - 200) + "px";

    document.body.appendChild(cat);

    cat.addEventListener("click", () => {
        score += catData.points;
        scoreBox.textContent = "Score: " + score;
        cat.remove();

        // Jumpscare trigger
        if (score >= 101) {
            window.location.href = "jumpscare.html";
        }
    });

    // Remove after 2 sec
    setTimeout(() => {
        if (cat.parentNode) cat.remove();
    }, 2000);
}

// Fast spawn
setInterval(spawnCat, 400);
</script>
